enum

#include<stdio.h>
// default start is 0
enum Week { Sun, Mon, Tue};

int main(void) {
  enum Week week;
  week = Sun;
  // Can not used other than defined enum value
  // week = Fri; // Error
  printf("%d", week);

  return 0;
}

"for" and "while" loop

#include<stdio.h>
for(int i = 0; i < 5; i++) {
  printf("for loop : %d\n", i);
}

int j = 0;
while(j == 3) {
  j++;
  if(j == 1) continue;
  printf("while loop : %d\n", j);
}

int k = 0;
do {
  k++;
  printf("%d\n", k);
}while(k == 5);

"switch case" and "goto"

#include<stdio.h>
int j = 2
bar:
swtich(j) {
    case 1: break;
    case 2: break;
    case 3: goto foo;
    case 4: goto bar;
    default: break;
}
foo:

Be careful not to become spaghetti code when using 'goto'.
